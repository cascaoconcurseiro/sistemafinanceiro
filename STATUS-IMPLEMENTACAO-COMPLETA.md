```typescript
import { IntegratedFinancialService } from '@/lib/services/integrated-financial-service';

// No seu componente ou contexto:
const createTransaction = async (data: any) => {
  try {
    const result = await IntegratedFinancialService.createTransaction(
      data,
      user.id,
      user.email
    );
    
    // Sucesso! Transação criada com:
    // ✅ Idempotência
    // ✅ Validações
    // ✅ Eventos
    // ✅ Auditoria
    
    return result;
  } catch (error) {
    // Tratamento de erro
    console.error('Erro ao criar transação:', error);
    throw error;
  }
};

// Transferência
const transfer = async (from, to, amount) => {
  return await IntegratedFinancialService.createTransfer({
    fromAccountId: from,
    toAccountId: to,
    amount,
    description: 'Transferência',
    date: new Date(),
    userId: user.id,
    userEmail: user.email
  });
};

// Pagar fatura
const payInvoice = async (invoiceId, accountId) => {
  return await IntegratedFinancialService.payInvoice(
    invoiceId,
    accountId,
    user.id,
    user.email
  );
};

// Relatórios
const getDRE = async () => {
  const startDate = new Date(2025, 0, 1); // Janeiro
  const endDate = new Date(2025, 11, 31); // Dezembro
  
  return await IntegratedFinancialService.generateDRE(
    user.id,
    startDate,
    endDate
  );
};
```
